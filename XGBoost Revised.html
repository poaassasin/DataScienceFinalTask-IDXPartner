def to_months(col):
    dates = pd.to_datetime(col, format='%b-%y', errors='coerce')
    ref_date = pd.to_datetime('2016-01-01')
    # Hitung selisih hari ('D'), lalu bagi 30 untuk estimasi bulan
    diff_days = (ref_date - dates).dt.days
    return (diff_days / 30).fillna(0).astype(int)

def apply_woe(df, feature, target):
    # Hitung distribusi Good dan Bad per kategori
    woe_df = df.groupby(feature)[target].agg(['count', 'sum'])
    woe_df['bad'] = woe_df['sum']
    woe_df['good'] = woe_df['count'] - woe_df['bad']
    
    # Hitung persentase
    total_good = woe_df['good'].sum()
    total_bad = woe_df['bad'].sum()
    
    woe_df['dist_good'] = woe_df['good'] / total_good
    woe_df['dist_bad'] = woe_df['bad'] / total_bad
    
    # Hitung WoE (tambah 0.001 agar tidak dibagi nol)
    woe_df['woe'] = np.log((woe_df['dist_good'] + 0.001) / (woe_df['dist_bad'] + 0.001))
    
    # Map kembali ke dataframe asli
    return df[feature].map(woe_df['woe'])

# Terapkan WoE pada fitur kategori utama daripada OHE
cat_features = ['grade', 'sub_grade', 'home_ownership', 'verification_status', 'purpose', 'addr_state']

    X, y, test_size=0.2, random_state=42, stratify=y
)

# Gunakan Scale Pos Weight biar Recall Bad Loan tinggi
ratio = (y == 0).sum() / (y == 1).sum()

xgb_legit = XGBClassifier(
    n_estimators=100,
    max_depth=7,
    learning_rate=0.05,
    scale_pos_weight=ratio, 
    tree_method='hist',
