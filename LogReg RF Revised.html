def to_months(col):
    dates = pd.to_datetime(col, format='%b-%y', errors='coerce')
    ref_date = pd.to_datetime('2016-01-01')
    # Hitung selisih hari ('D'), lalu bagi 30 untuk estimasi bulan
    diff_days = (ref_date - dates).dt.days
    return (diff_days / 30).fillna(0).astype(int)

def apply_woe(df, feature, target):
    # Hitung distribusi Good dan Bad per kategori
    woe_df = df.groupby(feature)[target].agg(['count', 'sum'])
    woe_df['bad'] = woe_df['sum']
    woe_df['good'] = woe_df['count'] - woe_df['bad']
    
    # Hitung persentase
    total_good = woe_df['good'].sum()
    total_bad = woe_df['bad'].sum()
    
    woe_df['dist_good'] = woe_df['good'] / total_good
    woe_df['dist_bad'] = woe_df['bad'] / total_bad
    
    # Hitung WoE (tambah 0.001 agar tidak dibagi nol)
    woe_df['woe'] = np.log((woe_df['dist_good'] + 0.001) / (woe_df['dist_bad'] + 0.001))
    
    # Map kembali ke dataframe asli
    return df[feature].map(woe_df['woe'])

# Terapkan WoE pada fitur kategori utama daripada OHE
cat_features = ['grade', 'sub_grade', 'home_ownership', 'verification_status', 'purpose', 'addr_state']

from sklearn.metrics import precision_recall_curve

def plot_threshold_curve(model, X_eval, model_name):
    y_probs = model.predict_proba(X_eval)[:, 1]
    precisions, recalls, thresholds = precision_recall_curve(y_test, y_probs)
    
    plt.figure(figsize=(10, 6))
    plt.plot(thresholds, precisions[:-1], 'b--', label='Precision')
    plt.plot(thresholds, recalls[:-1], 'g-', label='Recall')
    plt.axvline(x=0.5, color='red', linestyle=':', label='Default Threshold (0.5)')
    plt.xlabel('Threshold (Batas Probabilitas)')
    plt.ylabel('Score')
    plt.title(f'Precision-Recall Trade-off: {model_name}')
    plt.legend()
    plt.grid(True)
    plt.show()

from imblearn.over_sampling import SMOTE
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import roc_auc_score, classification_report
import pandas as pd
import matplotlib.pyplot as plt

# 1. Implementasi SMOTE pada Data Training (Scaled)
smote = SMOTE(random_state=42)
X_train_smote, y_train_smote = smote.fit_resample(X_train_scaled, y_train)

print(f"Jumlah data setelah SMOTE: {len(y_train_smote)}")
print(f"Distribusi Kelas: \n{pd.Series(y_train_smote).value_counts()}")

# 2. Training Model A: LogReg dengan class_weight='balanced' (Data Asli)
logreg_balanced = LogisticRegression(class_weight='balanced', max_iter=1000, random_state=42)
logreg_balanced.fit(X_train_scaled, y_train)

# 3. Training Model B: LogReg dengan SMOTE (Data Resampled, tanpa class_weight)
logreg_smote = LogisticRegression(max_iter=1000, random_state=42)
logreg_smote.fit(X_train_smote, y_train_smote)

# 4. Evaluasi Keduanya pada Data Test yang SAMA
models = {
    'LogReg (Balanced)': (logreg_balanced, X_test_scaled),
    'LogReg (SMOTE)': (logreg_smote, X_test_scaled)
}

results = []

for name, (model, X_eval) in models.items():
    y_prob = model.predict_proba(X_eval)[:, 1]
    y_pred = model.predict(X_eval)
    
    auc_score = roc_auc_score(y_test, y_prob)
    # Ambil recall untuk kelas 1 (Bad Loan)
    report = classification_report(y_test, y_pred, output_dict=True)
    recall_bad = report['1']['recall']
    precision_bad = report['1']['precision']
    
    results.append({
        'Model': name,
        'ROC-AUC': auc_score,
        'Recall (Bad)': recall_bad,
        'Precision (Bad)': precision_bad
    })
    
    print(f"\n--- {name} Classification Report ---")
    print(classification_report(y_test, y_pred))

# 5. Tampilkan Tabel Perbandingan
df_compare = pd.DataFrame(results)
print("\n--- TABEL PERBANDINGAN FINAL ---")
print(df_compare)

# 6. Visualisasi Sederhana
df_compare.set_index('Model')[['ROC-AUC', 'Recall (Bad)']].plot(kind='bar', figsize=(10, 5))
plt.title('LogReg Balanced vs SMOTE')
plt.ylabel('Score')
plt.xticks(rotation=0)
plt.legend(loc='lower right')
plt.show()
