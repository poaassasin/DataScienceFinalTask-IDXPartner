from sklearn.metrics import precision_recall_curve
import numpy as np
import matplotlib.pyplot as plt

# 1. Pastikan probabilitas sudah siap (Data berasal dari model Legit)
y_pred_proba = logreg.predict_proba(X_test_scaled)[:, 1] #

# 2. Hitung Precision dan Recall (Menghindari NameError)
precision, recall, thresholds = precision_recall_curve(y_test, y_pred_proba) #

# 3. Mencari index persimpangan secara otomatis
idx_list = np.argwhere(np.diff(np.sign(precision[:-1] - recall[:-1]))).flatten()
idx = idx_list[0] if len(idx_list) > 0 else 0 

# 4. Visualisasi untuk Infografis
plt.figure(figsize=(10, 6))
plt.plot(thresholds, precision[:-1], label='Precision', color='blue', lw=2)
plt.plot(thresholds, recall[:-1], label='Recall', color='green', lw=2)

# Garis ambang batas default (0.5)
plt.axvline(x=0.5, color='red', linestyle='--', alpha=0.6, label='Default Threshold (0.5)') #

# Garis persimpangan (Optimal Threshold)
plt.axvline(x=thresholds[idx], color='purple', linestyle='-.', 
            label=f'Intersection (~{thresholds[idx]:.2f})')
plt.scatter(thresholds[idx], precision[idx], color='black', s=100, zorder=5) 

plt.title('Precision-Recall Trade-off: Menentukan Batas Risiko')
plt.xlabel('Threshold (Ambang Batas Keputusan)')
plt.ylabel('Score')
plt.legend()
plt.grid(alpha=0.3)
plt.show()
