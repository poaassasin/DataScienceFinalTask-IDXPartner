import pandas as pd
import numpy as np
import xgboost as xgb
from xgboost import XGBClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report, roc_auc_score, confusion_matrix, ConfusionMatrixDisplay
import matplotlib.pyplot as plt
import gc

# ==========================================
# 1. LOAD DATASET & DEFINISI TARGET
# ==========================================
# low_memory=False untuk menangani DtypeWarning pada dataset besar
df = pd.read_csv('loan_data_2007_2014.csv', low_memory=False)

# Definisi Bad Loan (Target = 1)
bad_status = [
    'Charged Off', 'Default', 'Does not meet the credit policy. Status:Charged Off',
    'Late (31-120 days)'
]
df['target'] = np.where(df['loan_status'].isin(bad_status), 1, 0)

# ==========================================
# 2. PEMBERSIHAN DATA (OPTIMASI MEMORI)
# ==========================================
# A. Hapus ID & Fitur High-Cardinality (Penyebab proses lambat/30 menit)
high_cardinality = ['id', 'member_id', 'url', 'desc', 'title', 'zip_code', 'addr_state', 'emp_title', 'loan_status']
df = df.drop(columns=high_cardinality, errors='ignore')

# B. Hapus Fitur Leakage (Agar model tetap "Legit" & valid untuk nasabah baru)
leakage_cols = [
    'recoveries', 'collection_recovery_fee', 'total_rec_prncp', 
    'total_rec_int', 'total_rec_late_fee', 'last_pymnt_amnt', 
    'out_prncp', 'out_prncp_inv', 'total_pymnt', 'total_pymnt_inv',
    'last_pymnt_d', 'next_pymnt_d', 'last_credit_pull_d'
]
df = df.drop(columns=leakage_cols, errors='ignore')

# C. Log-Transformation pada annual_inc (Normalisasi distribusi)
df['annual_inc'] = np.log1p(df['annual_inc'].fillna(df['annual_inc'].median()))

# ==========================================
# 3. ENCODING, CLEANING & SPLITTING
# ==========================================
# One-Hot Encoding pada fitur sisa
X_encoded = pd.get_dummies(df.drop(columns=['target']), drop_first=True)

# --- FIX ValueError: Membersihkan nama kolom dari karakter [, ], atau < ---
X_encoded.columns = (
    X_encoded.columns
    .str.replace('[', '(', regex=False)
    .str.replace(']', ')', regex=False)
    .str.replace('<', 'less_than', regex=False)
)
# --------------------------------------------------------------------------

y = df['target']

# Bebaskan RAM
del df
gc.collect()

X_train, X_test, y_train, y_test = train_test_split(
    X_encoded, y, test_size=0.2, random_state=42, stratify=y
)

# ==========================================
# 4. TRAINING XGBOOST (MODE CEPAT)
# ==========================================
# Hitung rasio untuk scale_pos_weight agar Recall tinggi
ratio = (y == 0).sum() / (y == 1).sum()

xgb_model = XGBClassifier(
    n_estimators=100,
    max_depth=6,
    learning_rate=0.1,
    scale_pos_weight=ratio, 
    tree_method='hist',   # AKSELERASI: Menggunakan teknik histogram
    n_jobs=-1,            # Menggunakan seluruh core CPU laptop
    random_state=42,
    eval_metric='logloss'
)

xgb_model.fit(X_train, y_train)

# ==========================================
# 5. EVALUASI HASIL AKHIR
# ==========================================
y_prob = xgb_model.predict_proba(X_test)[:, 1]
y_pred = xgb_model.predict(X_test)

print(f"\n✅ SUCCESS! XGBoost ROC-AUC: {roc_auc_score(y_test, y_prob):.4f}")
print("\nClassification Report:")
print(classification_report(y_test, y_pred))

# Plot Confusion Matrix untuk perbandingan di PPT
cm = confusion_matrix(y_test, y_pred)
ConfusionMatrixDisplay(cm, display_labels=['Good', 'Bad']).plot(cmap='Purples')
plt.title('Confusion Matrix: XGBoost (Optimized & Legit)')
plt.show()

from sklearn.model_selection import RandomizedSearchCV

# 1. Tentukan Ruang Parameter (Grid)
param_grid = {
    'n_estimators': [100, 300, 500],
    'max_depth': [3, 5, 7, 9],
    'learning_rate': [0.01, 0.05, 0.1],
    'subsample': [0.7, 0.8, 0.9],
    'colsample_bytree': [0.7, 0.8, 0.9],
    'gamma': [0, 0.1, 0.2]
}

# 2. Inisialisasi RandomizedSearch
# n_iter=5 agar tidak terlalu lama (30 menit+), tetap gunakan tree_method='hist'
random_search = RandomizedSearchCV(
    estimator=XGBClassifier(
        scale_pos_weight=ratio, 
        tree_method='hist', 
        n_jobs=-1, 
        random_state=42, 
        eval_metric='logloss'
    ),
    param_distributions=param_grid,
    n_iter=5, 
    scoring='roc_auc',
    cv=3, 
    verbose=2,
    random_state=42
)

print("--- Memulai Proses Tuning (Estimasi 5-10 menit) ---")
random_search.fit(X_train, y_train)

# 3. Ambil Model Terbaik
best_xgb = random_search.best_estimator_
print(f"\n✅ Best Parameters: {random_search.best_params_}")

# 4. Evaluasi Final Model Tuned
y_prob_tuned = best_xgb.predict_proba(X_test)[:, 1]
y_pred_tuned = best_xgb.predict(X_test)

print(f"\n🚀 Tuned XGBoost ROC-AUC: {roc_auc_score(y_test, y_prob_tuned):.4f}")
print("\nClassification Report (Tuned):")
print(classification_report(y_test, y_pred_tuned))
